150 async def check(self, recorder: KafkaRecorder, endpoint: str) -> None: 

151 """ 

152 Runs a new HTTP(s) request on the configured URL and records probing 

153 results in a :class:`aiven_monitor.Measure`. 

154 

155 HTTP/1.1 and HTTP/2 are both supported. 

156 

157 :param recorder: A recorder to handle the Measure. 

158 :param endpoint: A name for the place from where this check is running. 

159 """ 

160 # While sharing a Client over many probes would be more efficient, it's 

161 # probably better to not skew the results by using pooled connections 

162 # between distinct probes and between runs of the same probe. 

163 async with httpx.AsyncClient( 

164 http2=True, headers={'user-agent': USER_AGENT} 

165 ) as client: 

166 logger.info('probe-check: %s', self) 

167 expected_pattern = ( 

168 None 

169 if self.expected_pattern is None 

170 else self.expected_pattern.pattern 

171 ) 

172 measure = Measure( 

173 self.url, 

174 uuid4(), 

175 endpoint, 

176 datetime.datetime.utcnow(), 

177 expected_pattern, 

178 ) 

179 with trio.move_on_after(REQUEST_TIMEOUT_SECS): 

180 async with client.stream('GET', self.url) as response: 

181 measure.protocol = response.http_version 

182 measure.status_code = response.status_code 

183 chunks = [] 

184 response_size_bytes = 0 

185 async for chunk in response.aiter_bytes(): 185 ↛ 200line 185 didn't jump to line 200, because the loop on line 185 didn't complete

186 # Make sure trio has the opportunity to check timeouts 

187 await trio.sleep(0) 

188 # We only record the content if we need it 

189 allowed_bytes = RESPONSE_MAX_BYTES - response_size_bytes 

190 expects_pattern = self.expected_pattern is not None 

191 if expects_pattern and allowed_bytes > 0: 

192 # Make sure we stop at RESPONSE_MAX_BYTES without 

193 # being affected by the exact size of each chunk. 

194 # Beware, we wil cut a multi-byte character in half. 

195 chunk = chunk[:allowed_bytes] 

196 chunks.append(chunk) 

197 response_size_bytes += len(chunk) 

198 if response_size_bytes >= RESPONSE_MAX_BYTES: 

199 break 

200 await response.aclose() 

201 redirections_time_secs = sum( 

202 r.elapsed.total_seconds() for r in response.history 

203 ) 

204 measure.response_time_secs = ( 

205 response.elapsed.total_seconds() 

206 + redirections_time_secs 

207 ) 

208 measure.response_size_bytes = response_size_bytes 

209 if self.expected_pattern is not None: 

210 try: 

211 # I wish we didn't call protected httpx code here 

212 # but it's better than rewriting and testing their 

213 # decoder, and this separates the concerns of 

214 # handling large vs. malformed responses. 

215 text_chunks = [] 

216 decoder = TextDecoder(encoding=response.encoding) 

217 for chunk in chunks: 

218 text_chunks.append(decoder.decode(chunk)) 

219 text_chunks.append(decoder.flush()) 

220 except ValueError as value_error: 

221 # TODO: Put that in the measure instead 

222 logger.error('response-error: %s', str(value_error)) 

223 else: 

224 content = ''.join(text_chunks) 

225 match = self.expected_pattern.search(content) 

226 measure.pattern_was_found = match is not None 

227 await recorder.record(measure)

230class Scheduler: 

231 """ 

232 Elects which :class:`Probe` object must be run next. 

233 

234 Internally, each added :class:`Probe` is wrapped in a :class:`Schedule`. 

235 """ 

236 

237 def __init__(self): 

238 self.queue = [] 

239 self.index = {} 

240 self.queue_changed = trio.Condition() 

241 

242 async def add(self, probes: Iterable[Probe]) -> None: 

243 """ 

244 Adds a list of probes. 

245 

246 Each one of them will be scheduled to be run somewhere between 

247 :func:`trio.current_time` and :func:`trio.current_time` + 

248 :attr:`Probe.interval_secs`. 

249 """ 

250 for probe in probes: 

251 logger.info('probe-add: %s', probe) 

252 schedule = Schedule(probe, trio.current_time()) 

253 self.index[probe] = schedule 

254 heapq.heappush(self.queue, schedule) 

255 async with self.queue_changed: 

256 self.queue_changed.notify() 

257 

258 async def remove(self, probes: Iterable[Probe]) -> None: 

259 """ 

260 Removes a list of probes. 

261 

262 It is not an error to try to remove a probe that was not previously 

263 added. 

264 """ 

265 for probe in probes: 

266 schedule = self.index.pop(probe, None) 

267 if schedule is not None: 

268 logger.info('probe-remove: %s', schedule.probe) 

269 # We mark a schedule as removed here and ignore it later 

270 schedule.removed = True 

271 async with self.queue_changed: 

272 self.queue_changed.notify() 

273 

274 async def get_next(self) -> Probe: 

275 """ 

276 Returns the next probe that must be run immediately. 

277 

278 If the earliest available probe needs to be run later, this will 

279 internally wait until it's time to run the probe. 

280 

281 If the scheduling queue changes while waiting, the earliest scheduled 

282 probe will be re-elected immediately. This means that you can 

283 concurrently add and remove probes and not wait until it's time to run 

284 the previously elected probe. 

285 

286 If the scheduling queue is empty, this will wait until something is 

287 added to the queue. 

288 """ 

289 schedule: Optional[Schedule] = None 

290 while schedule is None: 

291 try: 

292 # Try to get the earliest scheduled probe 

293 schedule = self.queue[0] 

294 except IndexError: 

295 # If there is none, wait for a change 

296 async with self.queue_changed: 

297 await self.queue_changed.wait() 

298 else: 

299 # Wait until it's time to run the scheduled probe 

300 with trio.move_on_at(schedule.next_time): 

301 # However, if the queue changes before it's time to run, 

302 # we forget the selected schedule to re-elect a new one. 

303 async with self.queue_changed: 

304 await self.queue_changed.wait() 

305 schedule = None 

306 # Just before running it, check if it's not actually removed 

307 if schedule is not None and schedule.removed: 

308 heapq.heappop(self.queue) 

309 schedule = None 

310 # Immediately reschedule the next run of the selected probe 

311 schedule.advance() 

312 heapq.heapreplace(self.queue, schedule) 

313 # Then let the caller actually run the elected probe 

314 return schedule.probe

317class Schedule: 

318 """ 

319 Maintains the scheduling state of each :class:`Probe` inside a 

320 :class:`Scheduler`. 

321 

322 The :class:`Scheduler` picks the :class:`Schedule` with the least 

323 :attr:`next_time` first. 

324 """ 

325 

326 def __init__(self, probe: Probe, base_time: float): 

327 self.probe: Probe = probe 

328 """The scheduled :class:`Probe`.""" 

329 self.base_time: float = base_time 

330 """The earliest time the :class:`Probe` is allowed to be scheduled. 

331  

332 This changes each time :func:`advance` is called.""" 

333 self.next_time: float = self._generate_next_time() 

334 """The actual time the :class:`Probe` is scheduled. 

335  

336 This is always between :attr:`base_time` and 

337 :attr:`base_time` + :attr:`Probe.interval_secs`.""" 

338 # This is used by the Scheduler to lazily remove Probes. 

339 self.removed: bool = False 

340 

341 def __lt__(self, other: Schedule) -> bool: 

342 """ 

343 Returns True if `self` should be scheduled before `other` and False 

344 otherwise. 

345 

346 Schedule are only compared according to their :attr:`next_time` 

347 property. 

348 """ 

349 return self.next_time.__lt__(other.next_time) 

350 

351 def advance(self) -> None: 

352 """ 

353 - Advances :attr:`base_time` by :attr:`Probe.interval_secs`. 

354 - Updates :attr:`next_time` to stay between :attr:`base_time` and 

355 :attr:`base_time` + :attr:`Probe.interval_secs`. 

356 """ 

357 self.base_time += self.probe.interval_secs 

358 self.next_time = self._generate_next_time() 

359 

360 def _generate_next_time(self): 

361 return self.base_time + self.probe.interval_secs * random.uniform(0, 1)

402 async def start(self) -> None: 

403 """ 

404 Starts the Kafka producer. 

405 

406 If the connection cannot be established, it will endlessly retry while 

407 waiting `connect_interval_secs` between each attempt. 

408 """ 

409 while self.producer is None: 

410 try: 

411 self.producer = self._producer_factory(

415 ssl_keyfile=self.ssl_keyfile,

419 except kafka.errors.NoBrokersAvailable: 

420 await trio.sleep(self.connect_interval_secs) 

421 else: 

422 logger.info('kafka-ready: %s', self.producer) 

423 async with self.has_producer: 

424 self.has_producer.notify_all()

426 async def record(self, measure: Measure) -> None: 

427 """ 

428 Records a :class:`aiven_monitor.Measure` in Kafka. 

429 

430 This does not wait for the measure to be actually sent. 

431 

432 Because this requires a working Kafka connection, this function will 

433 not return until :func:`start` has completed. 

434

436 :func:`start`. However, this means that if :func:`record` is called as a 

437 coroutine and not directly awaited, pending measures and coroutines can 

438 accumulate in memory until the connection is established. 

439 

440 """ 

441 logger.info('measure-record: %s', measure.to_dict()) 

442 async with self.has_producer: 

443 while self.producer is None: 

444 await self.has_producer.wait() 

445 self.producer.send(self.topic, measure.to_dict())
